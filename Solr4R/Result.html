def self.types_for(hash, mix = Nuggets::String::CamelscoreMixin)
  constants.map { |const|
    if hash.key?(const.to_s.extend(mix).underscore!)
      mod = const_get(const)
      mod if mod.is_a?(Module)
    end
  }.compact
end
