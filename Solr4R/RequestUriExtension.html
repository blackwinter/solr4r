def with_params(params)
  self.params, query = params, [self.query].compact

  params.each { |key, value|
    key = URI.escape(key.to_s)
    Array(value).each { |val| query << "#{key}=#{CGI.escape(val.to_s)}" }
  }

  self.query = query.join('&') unless query.empty?

  self
end
