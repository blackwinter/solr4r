def process(docs = @batch, size = size())
  next_size = size.fdiv(STEP).ceil

  docs.each_slice(size) { |batch|
    client.add(batch, *@args, &@block).success? ? nil :
      size > 1 ? process(batch, next_size) : @failed.concat(batch)
  }
end
